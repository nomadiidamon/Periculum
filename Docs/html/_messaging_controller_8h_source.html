18/**
19 * A UObject controller that manages all gameplay message subscriptions and broadcasts for a given context object.
20 * Owns all listener handles and cleans them up automatically when StopListeningToAll is called.
21 * Intended to be owned by a UMessagingComponent or a UUserWidget — the owner provides world context and GC rooting.
22 */
23
24UCLASS(BlueprintType)

32 /// Initializes the messaging controller with the provided outer object and context object.

100 // --- Templated functions below cannot be UFUNCTIONs, but are the primary interface for sending and receiving messages. ---

102 /// <summary>
103 /// Broadcasts a message on the provided channel.
104 /// If the message derives from FPericulumMessage, MessageSender is stamped with the ContextObject
105 /// and WorldTimeSeconds is stamped with the current world time if not already set.
106 /// </summary>
107 /// <param name="MessageChannel">The gameplay tag channel to broadcast on.</param>
108 /// <param name="Message">The message struct to broadcast. Passed by value so base fields can be stamped before sending.</param>
109 template<typename FMessageType>
110 void BroadcastMessage(FGameplayTag MessageChannel, FMessageType Message)
111 {
112 if constexpr (std::is_base_of_v<FPericulumMessage, FMessageType>)
113 {
114 if (!Message.MessageSender.IsValid())
115 {
116 Message.MessageSender = ContextObject;
117 }
118 Message.WorldTimeSeconds = GetWorld() ? GetWorld()->GetTimeSeconds() : 0.0;
119 }
120
121 if (UGameplayMessageSubsystem* Subsystem = GetMessageSubsystem())
122 {
123 Subsystem->BroadcastMessage(MessageChannel, Message);
124 }
125 }
126
127 /// <summary>
128 /// Broadcasts a message on a per-actor instanced channel derived from BaseTag and TargetObject.
129 /// Only listeners that registered via ListenForMessageFromContext with the same BaseTag
130 /// on a controller whose ContextObject matches TargetObject will receive the message.
131 /// If the message derives from FPericulumMessage, MessageReceiver and MessageSender are stamped automatically.
132 /// </summary>
133 /// <param name="BaseTag">The base gameplay tag used to derive the instanced channel.</param>
134 /// <param name="TargetObject">The actor that identifies the instanced channel to broadcast on.</param>
135 /// <param name="Message">The message struct to broadcast.</param>
136 template<typename FMessageType>
137 void BroadcastMessageToObject(FGameplayTag BaseTag, AActor* TargetObject, FMessageType Message)
138 {
139 if (!TargetObject)
140 {
141 UE_LOG(LogTemp, Warning,
142 TEXT("UMessagingController::BroadcastMessageToObject — TargetObject is null. No message was sent."));
143 return;
144 }
145
146 const FGameplayTag MessageChannel = GetActorChannel(BaseTag, TargetObject);
147 if (!MessageChannel.IsValid()) return;
148
149 if constexpr (std::is_base_of_v<FPericulumMessage, FMessageType>)
150 {
151 Message.MessageReceiver = TargetObject;
152 if (!Message.MessageSender.IsValid()) Message.MessageSender = ContextObject;
153 Message.WorldTimeSeconds = GetWorld() ? GetWorld()->GetTimeSeconds() : 0.0;
154 }
155
156 if (UGameplayMessageSubsystem* Subsystem = GetMessageSubsystem())
157 {
158 Subsystem->BroadcastMessage(MessageChannel, Message);
159 }
160 }
161
162 /// <summary>
163 /// Registers a listener for messages of the given type on the provided channel.
164 /// The listener handle is stored internally and will be unregistered automatically when StopListeningToAll is called.
165 /// All other ListenFor variants route through this function.
166 /// </summary>
167 /// <param name="MessageChannel">The gameplay tag channel to listen on.</param>
168 /// <param name="Callback">The function to invoke when a matching message is received.</param>
169 template<typename FMessageType>
170 FGameplayMessageListenerHandle ListenForMessage(FGameplayTag MessageChannel, TFunction<void(FGameplayTag, const FMessageType&)> Callback)
171 {
172 UGameplayMessageSubsystem* Subsystem = GetMessageSubsystem();
173 if (!Subsystem)
174 {
175 UE_LOG(LogTemp, Warning,
176 TEXT("UMessagingController::ListenForMessage — subsystem unavailable. No listener was registered."));
177 return FGameplayMessageListenerHandle();
178 }
179
180 FGameplayMessageListenerHandle Handle =
181 Subsystem->RegisterListener<FMessageType>(MessageChannel, Callback);
182 RegisteredListeners.Add(Handle);
183 return Handle;
184 }
185
186 /// <summary>
187 /// Registers a listener on the instanced channel derived from this controller's ContextObject.
188 /// Only receives messages broadcast via BroadcastMessageToObject targeting the same object.
189 /// Requires the ContextObject to be an AActor, as instanced channels are derived from actor identity.
190 /// </summary>
191 /// <param name="BaseTag">The base gameplay tag used to derive the instanced channel.</param>
192 /// <param name="Callback">The function to invoke when a matching message is received.</param>
193 template<typename FMessageType>
194 FGameplayMessageListenerHandle ListenForMessageFromContext(FGameplayTag BaseTag, TFunction<void(FGameplayTag, const FMessageType&)> Callback)
195 {
196 AActor* ContextActor = Cast<AActor>(ContextObject.Get());
197 if (!ContextActor)
198 {
199 UE_LOG(LogTemp, Warning,
200 TEXT("UMessagingController::ListenForMessageFromContext — ContextObject is null or not an AActor. "
201 "Instanced channels require an AActor as the ContextObject. No listener was registered."));
202 return FGameplayMessageListenerHandle();
203 }
204
205 const FGameplayTag MessageChannel = GetActorChannel(BaseTag, ContextActor);
206 if (!MessageChannel.IsValid()) return FGameplayMessageListenerHandle();
207
208 return ListenForMessage<FMessageType>(MessageChannel, Callback);
209 }
210
211 /// <summary>
212 /// Registers a listener on a channel that filters incoming messages to only those
213 /// where MessageReceiver matches the provided object.
214 /// All messages on the channel are still delivered by the subsystem — filtering occurs in the callback.
215 /// Requires FMessageType to derive from FPericulumMessage.
216 /// </summary>
217 /// <param name="MessageChannel">The gameplay tag channel to listen on.</param>
218 /// <param name="ReceiverFilter">The UObject that MessageReceiver must match for the callback to fire.</param>
219 /// <param name="Callback">The function to invoke when a matching message is received.</param>
220 template<typename FMessageType>
221 FGameplayMessageListenerHandle ListenForMessageTargetingObject(FGameplayTag MessageChannel, UObject* ReceiverFilter, TFunction<void(FGameplayTag, const FMessageType&)> Callback)
222 {
223 TWeakObjectPtr<UObject> WeakFilter = ReceiverFilter;
224 return ListenForMessage<FMessageType>(MessageChannel,
225 [WeakFilter, Callback](FGameplayTag Tag, const FMessageType& Message)
226 {
227 if constexpr (std::is_base_of_v<FPericulumMessage, FMessageType>)
228 {
229 if (Message.MessageReceiver.Get() != WeakFilter.Get()) return;
230 }
231 Callback(Tag, Message);
232 });
233 }
234
235 /// <summary>
236 /// Registers a listener on a channel that filters incoming messages to only those
237 /// where MessageSender matches the provided object.
238 /// All messages on the channel are still delivered by the subsystem — filtering occurs in the callback.
239 /// Requires FMessageType to derive from FPericulumMessage.
240 /// </summary>
241 /// <param name="MessageChannel">The gameplay tag channel to listen on.</param>
242 /// <param name="SenderFilter">The UObject that MessageSender must match for the callback to fire.</param>
243 /// <param name="Callback">The function to invoke when a matching message is received.</param>
244 template<typename FMessageType>
245 FGameplayMessageListenerHandle ListenForMessageFromObject(FGameplayTag MessageChannel, UObject* SenderFilter, TFunction<void(FGameplayTag, const FMessageType&)> Callback)
246 {
247 TWeakObjectPtr<UObject> WeakFilter = SenderFilter;
248 return ListenForMessage<FMessageType>(MessageChannel,
249 [WeakFilter, Callback](FGameplayTag Tag, const FMessageType& Message)
250 {
251 if constexpr (std::is_base_of_v<FPericulumMessage, FMessageType>)
252 {
253 if (Message.MessageSender.Get() != WeakFilter.Get()) return;
254 }
255 Callback(Tag, Message);
256 });
257 }
